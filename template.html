function wmo(code) { if (code === 0) return { label: "Céu limpo", svg: ICONS.sun }; if (code === 1 || code === 2) return { label: "Parcialmente nublado", svg: ICONS.cloud }; if (code === 3) return { label: "Nublado", svg: ICONS.cloud }; if (code >= 51 && code <= 67) return { label: "Chuva", svg: ICONS.rain }; if (code >= 80) return { label: "Tempestade", svg: ICONS.storm }; return { label: "Tempo variável", svg: ICONS.cloud }; } window.render = function (data) { const current = wmo( data.weather.code ); // Header document.getElementById( "place" ).textContent = data.city; document.getElementById( "date" ).textContent = data.today; // Hero document.getElementById( "heroIcon" ).innerHTML = current.svg; document.getElementById( "temp" ).textContent = data.weather.temp; document.getElementById( "cond" ).textContent = current.label; document.getElementById( "sub" ).innerHTML = "Sensação " + data.weather.feels + "° • " + data.weather.rainChance + "% de chuva • Vento " + data.weather.wind + " km/h"; // Forecast let forecastHtml = ""; data.weather.forecast.forEach( f => { const icon = wmo( f.code ); forecastHtml += `
${f.label}
${icon.svg}
${f.max}°
${f.min}°
${ICONS.drop} ${f.pop}%
`; } ); document.getElementById( "forecast" ).innerHTML = forecastHtml; // Stay if ( data.stay.occupied ) { document.getElementById( "stayLine" ).innerHTML = ` Check-out

${data.stay.checkout}

Até ${data.house.checkoutTime} `; } else { document.getElementById( "stayLine" ).innerHTML = data.house.welcome; } // House document.getElementById( "wifi" ).textContent = data.house.wifiName; document.getElementById( "password" ).textContent = data.house.wifiPassword; document.getElementById( "checkoutTime" ).textContent = data.house.checkoutTime; // Footer document.getElementById( "updated" ).textContent = "Atualizado às " + data.updatedAt; };
